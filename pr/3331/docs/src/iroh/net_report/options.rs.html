6mod imp {
7    use std::{collections::BTreeSet, sync::Arc};
8
9    use netwatch::UdpSocket;
10
11    use crate::net_report::{reportgen::ProbeProto, QuicConfig};
12
13    /// Options for running probes
14    ///
15    /// By default, will run icmp over IPv4, icmp over IPv6, and Https probes.
16    ///
17    /// Use [`Options::stun_v4`], [`Options::stun_v6`], and [`Options::quic_config`]
18    /// to enable STUN over IPv4, STUN over IPv6, and QUIC address discovery.
19    #[derive(Debug, Clone)]
20    pub struct Options {
21        /// Socket to send IPv4 STUN probes from.
22        ///
23        /// Responses are never read from this socket, they must be passed in via internal
24        /// messaging since, when used internally in iroh, the socket is also used to receive
25        /// other packets from in the magicsocket (`MagicSock`).
26        ///
27        /// If not provided, STUN probes will not be sent over IPv4.
28        pub(crate) stun_sock_v4: Option<Arc<UdpSocket>>,
29        /// Socket to send IPv6 STUN probes from.
30        ///
31        /// Responses are never read from this socket, they must be passed in via internal
32        /// messaging since, when used internally in iroh, the socket is also used to receive
33        /// other packets from in the magicsocket (`MagicSock`).
34        ///
35        /// If not provided, STUN probes will not be sent over IPv6.
36        pub(crate) stun_sock_v6: Option<Arc<UdpSocket>>,
37        /// The configuration needed to launch QUIC address discovery probes.
38        ///
39        /// If not provided, will not run QUIC address discovery.
40        pub(crate) quic_config: Option<QuicConfig>,
41        /// Enable icmp_v4 probes
42        ///
43        /// On by default
44        pub(crate) icmp_v4: bool,
45        /// Enable icmp_v6 probes
46        ///
47        /// On by default
48        pub(crate) icmp_v6: bool,
49        /// Enable https probes
50        ///
51        /// On by default
52        pub(crate) https: bool,
53    }
54
55    impl Default for Options {
56        fn default() -> Self {
57            Self {
58                stun_sock_v4: None,
59                stun_sock_v6: None,
60                quic_config: None,
61                icmp_v4: true,
62                icmp_v6: true,
63                https: true,
64            }
65        }
66    }
67
68    impl Options {
69        /// Create an [`Options`] that disables all probes
70        pub fn disabled() -> Self {
71            Self {
72                stun_sock_v4: None,
73                stun_sock_v6: None,
74                quic_config: None,
75                icmp_v4: false,
76                icmp_v6: false,
77                https: false,
78            }
79        }
80
81        /// Set the ipv4 stun socket and enable ipv4 stun probes
82        pub fn stun_v4(mut self, sock: Option<Arc<UdpSocket>>) -> Self {
83            self.stun_sock_v4 = sock;
84            self
85        }
86
87        /// Set the ipv6 stun socket and enable ipv6 stun probes
88        pub fn stun_v6(mut self, sock: Option<Arc<UdpSocket>>) -> Self {
89            self.stun_sock_v6 = sock;
90            self
91        }
92
93        /// Enable quic probes
94        pub fn quic_config(mut self, quic_config: Option<QuicConfig>) -> Self {
95            self.quic_config = quic_config;
96            self
97        }
98
99        /// Enable or disable icmp_v4 probe
100        pub fn icmp_v4(mut self, enable: bool) -> Self {
101            self.icmp_v4 = enable;
102            self
103        }
104
105        /// Enable or disable icmp_v6 probe
106        pub fn icmp_v6(mut self, enable: bool) -> Self {
107            self.icmp_v6 = enable;
108            self
109        }
110
111        /// Enable or disable https probe
112        pub fn https(mut self, enable: bool) -> Self {
113            self.https = enable;
114            self
115        }
116
117        /// Turn the options into set of valid protocols
118        pub(crate) fn to_protocols(&self) -> BTreeSet<ProbeProto> {
119            let mut protocols = BTreeSet::new();
120            if self.stun_sock_v4.is_some() {
121                protocols.insert(ProbeProto::StunIpv4);
122            }
123            if self.stun_sock_v6.is_some() {
124                protocols.insert(ProbeProto::StunIpv6);
125            }
126            if let Some(ref quic) = self.quic_config {
127                if quic.ipv4 {
128                    protocols.insert(ProbeProto::QuicIpv4);
129                }
130                if quic.ipv6 {
131                    protocols.insert(ProbeProto::QuicIpv6);
132                }
133            }
134            if self.icmp_v4 {
135                protocols.insert(ProbeProto::IcmpV4);
136            }
137            if self.icmp_v6 {
138                protocols.insert(ProbeProto::IcmpV6);
139            }
140            if self.https {
141                protocols.insert(ProbeProto::Https);
142            }
143            protocols
144        }
145    }
146}

164    }
165

168            Self { https: true }
169        }
170    }
171
172    impl Options {
173        /// Create an [`Options`] that disables all probes
174        pub fn disabled() -> Self {
175            Self { https: false }
176        }
177
178        /// Enable or disable https probe
179        pub fn https(mut self, enable: bool) -> Self {
180            self.https = enable;
181            self
182        }
183
184        /// Turn the options into set of valid protocols
185        pub(crate) fn to_protocols(&self) -> BTreeSet<ProbeProto> {
186            let mut protocols = BTreeSet::new();
187            if self.https {
188                protocols.insert(ProbeProto::Https);
