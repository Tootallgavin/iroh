70/// The maximum latency of all nodes, if none are found yet.
71///
72/// Normally the max latency of all nodes is computed, but if we don't yet know any nodes
73/// latencies we return this as default.  This is the value of the initial STUN probe
74/// delays.  It is only used as time to wait for further latencies to arrive, which *should*
75/// never happen unless there already is at least one latency.  Yet here we are, defining a
76/// default which will never be used.
77const DEFAULT_MAX_LATENCY: Duration = Duration::from_millis(100);

79const ENOUGH_NODES: usize = 3;

81/// Client to run net_reports.
82///
83/// Creating this creates a net_report actor which runs in the background.  Most of the time
84/// it is idle unless [`Client::get_report`] is called, which is the main interface.
85///
86/// The [`Client`] struct can be cloned and results multiple handles to the running actor.
87/// If all [`Client`]s are dropped the actor stops running.
88///
89/// While running the net_report actor expects to be passed all received stun packets using
90/// `Addr::receive_stun_packet`.
91#[derive(Debug)]
92pub(crate) struct Client {
93    #[cfg(not(wasm_browser))]
94    socket_state: SocketState,
95    metrics: Arc<Metrics>,
96    probes: BTreeSet<Probe>,
97    relay_map: RelayMap,
98    #[cfg(any(test, feature = "test-utils"))]
99    insecure_skip_relay_cert_verify: bool,
100
101    /// A collection of previously generated reports.
102    ///
103    /// Sometimes it is useful to look at past reports to decide what to do.
104    reports: Reports,
105}
106
107#[derive(Debug)]
108struct Reports {
109    /// Do a full relay scan, even if last is `Some`.
110    next_full: bool,
111    /// Some previous reports.
112    prev: BTreeMap<Instant, Report>,
113    /// Most recent report.
114    last: Option<Report>,
115    /// Time of last full (non-incremental) report.
116    last_full: Instant,
117}
118
119impl Default for Reports {
120    fn default() -> Self {
121        Self {
122            next_full: true,
123            prev: Default::default(),
124            last: Default::default(),
125            last_full: Instant::now(),
126        }
127    }

167    /// Generates a [`Report`].
168    ///
169    /// Look at [`Options`] for the different configuration options.
170    pub(crate) async fn get_report(&mut self, if_state: IfStateDetails) -> Report {
171        debug!("net_report starting");
172        trace!(probes = ?self.probes, "Attempting probes");
173
174        let now = Instant::now();
175
176        let mut do_full = self.reports.next_full
177            || now.duration_since(self.reports.last_full) > FULL_REPORT_INTERVAL;
178
179        // If the last report had a captive portal and reported no UDP access,
180        // it's possible that we didn't get a useful net_report due to the
181        // captive portal blocking us. If so, make this report a full (non-incremental) one.
182        if !do_full {
183            if let Some(ref last) = self.reports.last {
184                do_full = !last.has_udp() && last.captive_portal.unwrap_or_default();
185            }
186        }
187        if do_full {
188            self.reports.last = None; // causes ProbePlan::new below to do a full (initial) plan
189            self.reports.next_full = false;
190            self.reports.last_full = now;
191            self.metrics.reports_full.inc();
192        }
193        self.metrics.reports.inc();
194
195        let enough_relays = std::cmp::min(self.relay_map.len(), ENOUGH_NODES);
196        #[cfg(wasm_browser)]
197        let if_state = IfStateDetails::default();
198        #[cfg(not(wasm_browser))]
199        let if_state = IfStateDetails {
200            have_v4: if_state.have_v4,
201            have_v6: if_state.have_v6,
202        };
203
204        let (actor, mut probe_rx) = reportgen::Client::new(
205            self.reports.last.clone(),
206            self.relay_map.clone(),
207            self.probes.clone(),
208            if_state,
209            #[cfg(not(wasm_browser))]
210            self.socket_state.clone(),
211            #[cfg(any(test, feature = "test-utils"))]
212            self.insecure_skip_relay_cert_verify,
213        );
214
215        let mut report = Report {
216            ..Default::default()
217        };
218
219        let mut timeout_fut = std::pin::pin!(MaybeFuture::default());
220
221        loop {
222            tokio::select! {
223                biased;
224
225                _ = &mut timeout_fut, if timeout_fut.is_some() => {
226                    drop(actor); // shuts down the probes
227                    break;
228                }
229
230                maybe_probe = probe_rx.recv() => {
231                    let Some(probe_res) = maybe_probe else {
232                        break;
233                    };
234                    trace!(?probe_res, "handling probe");
235                    match probe_res {
236                        ProbeFinished::Regular(probe) => match probe {
237                            Ok(probe) => {
238                                report.update(&probe);
239                                if timeout_fut.is_none() {
240                                    if let Some(timeout) = self.have_enough_reports(enough_relays, &report) {
241                                        timeout_fut.as_mut().set_future(time::sleep(timeout));
242                                    }
243                                }
244                            }
245                            Err(err) => {
246                                trace!("probe errored: {:?}", err);
247                            }
248                        },
249                        #[cfg(not(wasm_browser))]
250                        ProbeFinished::CaptivePortal(portal) => {
251                            report.captive_portal = portal;
252                        }
253                    }
254                }
255            }
256        }
257
258        self.add_report_history_and_set_preferred_relay(&mut report);
259        debug!("{report:?}");
260
261        report
262    }
263
264    fn have_enough_reports(&self, enough_relays: usize, report: &Report) -> Option<Duration> {
265        // Once we've heard from enough relay servers (3), start a timer to give up on the other
266        // probes. The timer's duration is a function of whether this is our initial full
267        // probe or an incremental one. For incremental ones, wait for the duration of the
268        // slowest relay. For initial ones, double that.
269        let latencies: Vec<Duration> = report.relay_latency.iter().map(|(_, l)| l).collect();
270        let have_enough_latencies = latencies.len() >= enough_relays;
271
272        if have_enough_latencies {
273            let timeout = latencies
274                .iter()
275                .max()
276                .copied()
277                .unwrap_or(DEFAULT_MAX_LATENCY);
278            let timeout = match self.reports.last.is_some() {
279                true => timeout,
280                false => timeout * 2,
281            };
282            debug!(
283                reports=latencies.len(),
284                delay=?timeout,
285                "Have enough probe reports, aborting further probes soon",
286            );
287
288            Some(timeout)
289        } else {
290            None
291        }
292    }

294    /// Adds `r` to the set of recent Reports and mutates `r.preferred_relay` to contain the best recent one.
295    fn add_report_history_and_set_preferred_relay(&mut self, r: &mut Report) {
296        let mut prev_relay = None;
297        if let Some(ref last) = self.reports.last {
298            prev_relay.clone_from(&last.preferred_relay);
299        }
300        let now = Instant::now();
301        const MAX_AGE: Duration = Duration::from_secs(5 * 60);
302
303        // relay ID => its best recent latency in last MAX_AGE
304        let mut best_recent = RelayLatencies::default();
305
306        // chain the current report as we are still mutating it
307        let prevs_iter = self
308            .reports
309            .prev
310            .iter()
311            .map(|(a, b)| -> (&Instant, &Report) { (a, b) });
312
313        let mut to_remove = Vec::new();
314        for (t, pr) in prevs_iter {
315            if now.duration_since(*t) > MAX_AGE {
316                to_remove.push(*t);
317                continue;
318            }
319            best_recent.merge(&pr.relay_latency);
320        }
321        // merge in current run
322        best_recent.merge(&r.relay_latency);
323
324        for t in to_remove {
325            self.reports.prev.remove(&t);
326        }
327
328        // Then, pick which currently-alive relay server from the
329        // current report has the best latency over the past MAX_AGE.
330        let mut best_any = Duration::default();
331        let mut old_relay_cur_latency = Duration::default();
332        {
333            for (url, duration) in r.relay_latency.iter() {
334                if Some(url) == prev_relay.as_ref() {
335                    old_relay_cur_latency = duration;
336                }
337                if let Some(best) = best_recent.get(url) {
338                    if r.preferred_relay.is_none() || best < best_any {
339                        best_any = best;
340                        r.preferred_relay.replace(url.clone());
341                    }
342                }
343            }
344
345            // If we're changing our preferred relay but the old one's still
346            // accessible and the new one's not much better, just stick with
347            // where we are.
348            if prev_relay.is_some()
349                && r.preferred_relay != prev_relay
350                && !old_relay_cur_latency.is_zero()
351                && best_any > old_relay_cur_latency / 3 * 2
352            {
353                r.preferred_relay = prev_relay;
354            }
355        }
356
357        self.reports.prev.insert(now, r.clone());
358        self.reports.last = Some(r.clone());
359    }

383    /// Create a [`crate::RelayMap`] of the given size.
384    ///
385    /// This function uses [`relay`]. Note that the returned map uses internal order that will
386    /// often _not_ match the order of the servers.
387    pub(crate) async fn relay_map(relays: usize) -> (Vec<server::Server>, crate::RelayMap) {
388        let mut servers = Vec::with_capacity(relays);
389        let mut nodes = Vec::with_capacity(relays);
390        for _ in 0..relays {
391            let (relay_server, node) = relay().await;
392            servers.push(relay_server);
393            nodes.push(node);
394        }
395        (servers, crate::RelayMap::from_iter(nodes))
396    }

499        }
