68/// The maximum latency of all nodes, if none are found yet.
69///
70/// Normally the max latency of all nodes is computed, but if we don't yet know any nodes
71/// latencies we return this as default.  This is the value of the initial STUN probe
72/// delays.  It is only used as time to wait for further latencies to arrive, which *should*
73/// never happen unless there already is at least one latency.  Yet here we are, defining a
74/// default which will never be used.
75const DEFAULT_MAX_LATENCY: Duration = Duration::from_millis(100);

77const ENOUGH_NODES: usize = 3;

79/// Client to run net_reports.
80///
81/// Creating this creates a net_report actor which runs in the background.  Most of the time
82/// it is idle unless [`Client::get_report`] is called, which is the main interface.
83///
84/// The [`Client`] struct can be cloned and results multiple handles to the running actor.
85/// If all [`Client`]s are dropped the actor stops running.
86///
87/// While running the net_report actor expects to be passed all received stun packets using
88/// `Addr::receive_stun_packet`.
89#[derive(Debug)]
90pub(crate) struct Client {
91    /// The port mapper client, if those are requested.
92    ///
93    /// The port mapper is responsible for talking to routers via UPnP and the like to try
94    /// and open ports.
95    #[cfg(not(wasm_browser))]
96    port_mapper: Option<portmapper::Client>,
97    /// The DNS resolver to use for probes that need to perform DNS lookups
98    #[cfg(not(wasm_browser))]
99    dns_resolver: DnsResolver,
100    /// The [`IpMappedAddresses`] that allows you to do QAD in iroh
101    #[cfg(not(wasm_browser))]
102    ip_mapped_addrs: Option<IpMappedAddresses>,
103    metrics: Arc<Metrics>,
104
105    /// A collection of previously generated reports.
106    ///
107    /// Sometimes it is useful to look at past reports to decide what to do.
108    reports: Reports,
109}
110
111#[derive(Debug)]
112struct Reports {
113    /// Do a full relay scan, even if last is `Some`.
114    next_full: bool,
115    /// Some previous reports.
116    prev: BTreeMap<Instant, Report>,
117    /// Most recent report.
118    last: Option<Report>,
119    /// Time of last full (non-incremental) report.
120    last_full: Instant,
121}
122
123impl Default for Reports {
124    fn default() -> Self {
125        Self {
126            next_full: true,
127            prev: Default::default(),
128            last: Default::default(),
129            last_full: Instant::now(),
130        }
131    }
132}

154    /// Generates a [`Report`].
155    ///
156    /// Look at [`Options`] for the different configuration options.
157    pub(crate) async fn get_report(
158        &mut self,
159        relay_map: RelayMap,
160        if_state: IfStateDetails,
161        opts: Options,
162    ) -> Report {
163        debug!("net_report starting");
164
165        let protocols = opts.to_protocols();
166        #[cfg(not(wasm_browser))]
167        let socket_state = SocketState {
168            port_mapper: self.port_mapper.clone(),
169            quic_config: opts.quic_config,
170            dns_resolver: self.dns_resolver.clone(),
171            ip_mapped_addrs: self.ip_mapped_addrs.clone(),
172        };
173        trace!("Attempting probes for protocols {protocols:#?}");
174
175        let now = Instant::now();
176
177        let mut do_full = self.reports.next_full
178            || now.duration_since(self.reports.last_full) > FULL_REPORT_INTERVAL;
179
180        // If the last report had a captive portal and reported no UDP access,
181        // it's possible that we didn't get a useful net_report due to the
182        // captive portal blocking us. If so, make this report a full (non-incremental) one.
183        if !do_full {
184            if let Some(ref last) = self.reports.last {
185                do_full = !last.udp && last.captive_portal.unwrap_or_default();
186            }
187        }
188        if do_full {
189            self.reports.last = None; // causes ProbePlan::new below to do a full (initial) plan
190            self.reports.next_full = false;
191            self.reports.last_full = now;
192            self.metrics.reports_full.inc();
193        }
194        self.metrics.reports.inc();
195
196        let enough_relays = std::cmp::min(relay_map.len(), ENOUGH_NODES);
197        #[cfg(wasm_browser)]
198        let if_state = IfStateDetails::default();
199        #[cfg(not(wasm_browser))]
200        let if_state = IfStateDetails {
201            have_v4: if_state.have_v4,
202            have_v6: if_state.have_v6,
203        };
204
205        let (actor, mut probe_rx) = reportgen::Client::new(
206            self.reports.last.clone(),
207            relay_map,
208            protocols,
209            if_state,
210            #[cfg(not(wasm_browser))]
211            socket_state,
212            #[cfg(any(test, feature = "test-utils"))]
213            opts.insecure_skip_relay_cert_verify,
214        );
215
216        let mut report = Report {
217            os_has_ipv6: os_has_ipv6(),
218            ..Default::default()
219        };
220
221        let mut timeout_fut = std::pin::pin!(MaybeFuture::default());
222
223        loop {
224            tokio::select! {
225                biased;
226
227                _ = &mut timeout_fut, if timeout_fut.is_some() => {
228                    drop(actor); // shuts down the probes
229                    break;
230                }
231
232                maybe_probe = probe_rx.recv() => {
233                    let Some(probe_res) = maybe_probe else {
234                        break;
235                    };
236                    trace!(?probe_res, "handling probe");
237                    match probe_res {
238                        ProbeFinished::Regular(probe) => match probe {
239                            Ok(probe) => {
240                                report.update(&probe);
241                                if timeout_fut.is_none() {
242                                    if let Some(timeout) = self.have_enough_reports(enough_relays, &report) {
243                                        timeout_fut.as_mut().set_future(time::sleep(timeout));
244                                    }
245                                }
246                            }
247                            Err(err) => {
248                                trace!("probe errored: {:?}", err);
249                            }
250                        },
251                        #[cfg(not(wasm_browser))]
252                        ProbeFinished::CaptivePortal(portal) => {
253                            report.captive_portal = portal;
254                        }
255                        #[cfg(not(wasm_browser))]
256                        ProbeFinished::Portmap(portmap) => {
257                            report.portmap_probe = portmap;
258                        }
259                    }
260                }
261            }
262        }
263
264        self.add_report_history_and_set_preferred_relay(&mut report);
265        debug!("{report:?}");
266
267        report
268    }
269
270    fn have_enough_reports(&self, enough_relays: usize, report: &Report) -> Option<Duration> {
271        // Once we've heard from enough relay servers (3), start a timer to give up on the other
272        // probes. The timer's duration is a function of whether this is our initial full
273        // probe or an incremental one. For incremental ones, wait for the duration of the
274        // slowest relay. For initial ones, double that.
275        let latencies: Vec<Duration> = report.relay_latency.iter().map(|(_, l)| l).collect();
276        let have_enough_latencies = latencies.len() >= enough_relays;
277
278        if have_enough_latencies {
279            let timeout = latencies
280                .iter()
281                .max()
282                .copied()
283                .unwrap_or(DEFAULT_MAX_LATENCY);
284            let timeout = match self.reports.last.is_some() {
285                true => timeout,
286                false => timeout * 2,
287            };
288            debug!(
289                reports=latencies.len(),
290                delay=?timeout,
291                "Have enough probe reports, aborting further probes soon",
292            );
293
294            Some(timeout)
295        } else {
296            None
297        }
298    }

300    /// Adds `r` to the set of recent Reports and mutates `r.preferred_relay` to contain the best recent one.
301    fn add_report_history_and_set_preferred_relay(&mut self, r: &mut Report) {
302        let mut prev_relay = None;
303        if let Some(ref last) = self.reports.last {
304            prev_relay.clone_from(&last.preferred_relay);
305        }
306        let now = Instant::now();
307        const MAX_AGE: Duration = Duration::from_secs(5 * 60);
308
309        // relay ID => its best recent latency in last MAX_AGE
310        let mut best_recent = RelayLatencies::default();
311
312        // chain the current report as we are still mutating it
313        let prevs_iter = self
314            .reports
315            .prev
316            .iter()
317            .map(|(a, b)| -> (&Instant, &Report) { (a, b) });
318
319        let mut to_remove = Vec::new();
320        for (t, pr) in prevs_iter {
321            if now.duration_since(*t) > MAX_AGE {
322                to_remove.push(*t);
323                continue;
324            }
325            best_recent.merge(&pr.relay_latency);
326        }
327        // merge in current run
328        best_recent.merge(&r.relay_latency);
329
330        for t in to_remove {
331            self.reports.prev.remove(&t);
332        }
333
334        // Then, pick which currently-alive relay server from the
335        // current report has the best latency over the past MAX_AGE.
336        let mut best_any = Duration::default();
337        let mut old_relay_cur_latency = Duration::default();
338        {
339            for (url, duration) in r.relay_latency.iter() {
340                if Some(url) == prev_relay.as_ref() {
341                    old_relay_cur_latency = duration;
342                }
343                if let Some(best) = best_recent.get(url) {
344                    if r.preferred_relay.is_none() || best < best_any {
345                        best_any = best;
346                        r.preferred_relay.replace(url.clone());
347                    }
348                }
349            }
350
351            // If we're changing our preferred relay but the old one's still
352            // accessible and the new one's not much better, just stick with
353            // where we are.
354            if prev_relay.is_some()
355                && r.preferred_relay != prev_relay
356                && !old_relay_cur_latency.is_zero()
357                && best_any > old_relay_cur_latency / 3 * 2
358            {
359                r.preferred_relay = prev_relay;
360            }
361        }
362
363        self.reports.prev.insert(now, r.clone());
364        self.reports.last = Some(r.clone());
365    }

386    pub(crate) async fn relay() -> (server::Server, RelayNode) {
387        let server = server::Server::spawn(server::testing::server_config())
388            .await
389            .expect("should serve relay");
390        let quic = Some(RelayQuicConfig {
391            port: server.quic_addr().expect("server should run quic").port(),
392        });
393        let node_desc = RelayNode {
394            url: server.https_url().expect("should work as relay"),
395            quic,
396        };
397
398        (server, node_desc)
399    }

401    /// Create a [`crate::RelayMap`] of the given size.
402    ///
403    /// This function uses [`relay`]. Note that the returned map uses internal order that will
404    /// often _not_ match the order of the servers.
405    pub(crate) async fn relay_map(relays: usize) -> (Vec<server::Server>, crate::RelayMap) {
406        let mut servers = Vec::with_capacity(relays);
407        let mut nodes = Vec::with_capacity(relays);
408        for _ in 0..relays {
409            let (relay_server, node) = relay().await;
410            servers.push(relay_server);
411            nodes.push(node);
412        }
413        (servers, crate::RelayMap::from_iter(nodes))
414    }
