1//! Actor to run hairpinning check.
2//!
3//! This actor works as follows:
4//!
5//! - After starting prepares the haircheck:
6//!   - binds socket
7//!   - sends traffic from it's socket to trick some routers
8//! - When requested performs the hairpin probe.
9//!   - result is sent to net_report actor addr.
10//! - Shuts down
11//!
12//! Note it will only perform a single hairpin check before shutting down.  Any further
13//! requests to it will fail which is intentional.
14

29/// Handle to the hairpin actor.
30///
31/// Dropping it will abort the actor.
32#[derive(Debug)]
33pub(super) struct Client {
34    addr: Option<oneshot::Sender<Message>>,
35    _drop_guard: AbortOnDropHandle<()>,
36}
37
38impl Client {
39    pub(super) fn new(net_report: net_report::Addr, reportgen: reportgen::Addr) -> Self {
40        let (addr, msg_rx) = oneshot::channel();
41
42        let actor = Actor {
43            msg_rx,
44            net_report,
45            reportgen,
46        };
47
48        let task =
49            task::spawn(async move { actor.run().await }.instrument(info_span!("hairpin.actor")));
50        Self {
51            addr: Some(addr),
52            _drop_guard: AbortOnDropHandle::new(task),
53        }
54    }
55
56    /// Returns `true` if we have started a hairpin check before.
57    pub(super) fn has_started(&self) -> bool {
58        self.addr.is_none()
59    }
60
61    /// Starts the hairpin check.
62    ///
63    /// *dst* should be our own address as discovered by STUN.  Hairpin detection works by
64    /// sending a new STUN request to our own public address, if we receive this request
65    /// back then hairpinning works, otherwise it does not.
66    ///
67    /// Will do nothing if this actor is already finished or a check has already started.
68    pub(super) fn start_check(&mut self, dst: SocketAddrV4) {
69        if let Some(addr) = self.addr.take() {
70            addr.send(Message::StartCheck(dst)).ok();
71        }
72    }
73}

75#[derive(Debug)]
76enum Message {
77    /// Performs the hairpin check.
78    ///
79    /// The STUN request will be sent to the provided [`SocketAddrV4`] which should be our
80    /// own address discovered using STUN.
81    StartCheck(SocketAddrV4),
82}
83
84#[derive(Debug)]
85struct Actor {
86    msg_rx: oneshot::Receiver<Message>,
87    net_report: net_report::Addr,
88    reportgen: reportgen::Addr,
89}
90
91#[derive(Debug, Snafu)]
92enum Error {
93    #[snafu(transparent)]
94    Io { source: std::io::Error },
95    #[snafu(display("net_report actor is gone"))]
96    NetReportActorGone,
97    #[snafu(display("reportgen actor is gone"))]
98    ReportGenActorGone,
99    #[snafu(display("stun response channel dropped"))]
100    StunResponseGone,
101}
102
103impl Actor {
104    async fn run(self) {
105        match self.run_inner().await {
106            Ok(_) => trace!("hairpin actor finished successfully"),
107            Err(err) => error!("Hairpin actor failed: {err:#}"),
108        }
109    }
110
111    async fn run_inner(self) -> Result<(), Error> {
112        let socket = UdpSocket::bind_v4(0)?;
113
114        if let Err(err) = Self::prepare_hairpin(&socket).await {
115            warn!("unable to send hairpin prep: {err:#}");
116            // Continue anyway, most routers are fine.
117        }
118
119        // We only have one message to handle
120        let Ok(Message::StartCheck(dst)) = self.msg_rx.await else {
121            return Ok(());
122        };
123
124        let txn = stun::TransactionId::default();
125        trace!(%txn, "Sending hairpin with transaction ID");
126        let (stun_tx, stun_rx) = oneshot::channel();
127        let inflight = Inflight {
128            txn,
129            start: Instant::now(), // ignored by hairping probe
130            s: stun_tx,
131        };
132        let (msg_response_tx, msg_response_rx) = oneshot::channel();
133        self.net_report
134            .send(net_report::Message::InFlightStun(inflight, msg_response_tx))
135            .await
136            .map_err(|_| NetReportActorGoneSnafu.build())?;
137
138        msg_response_rx
139            .await
140            .map_err(|_| NetReportActorGoneSnafu.build())?;
141
142        if let Err(err) = socket.send_to(&stun::request(txn), dst.into()).await {
143            warn!(%dst, "failed to send hairpin check");
144            return Err(err.into());
145        }
146
147        let now = Instant::now();
148        let hairpinning_works = match time::timeout(HAIRPIN_CHECK_TIMEOUT, stun_rx).await {
149            Ok(Ok(_)) => true,
150            Ok(Err(_)) => return Err(StunResponseGoneSnafu.build()),
151            Err(_) => false, // Elapsed
152        };
153        debug!(
154            "hairpinning done in {:?}, res: {:?}",
155            now.elapsed(),
156            hairpinning_works
157        );
158
159        self.reportgen
160            .send(super::Message::HairpinResult(hairpinning_works))
161            .await
162            .map_err(|_| ReportGenActorGoneSnafu.build())?;
163
164        trace!("reportgen notified");
165
166        Ok(())
167    }
168
169    async fn prepare_hairpin(socket: &UdpSocket) -> std::io::Result<()> {
170        // At least the Apple Airport Extreme doesn't allow hairpin
171        // sends from a private socket until it's seen traffic from
172        // that src IP:port to something else out on the internet.
173        //
174        // See https://github.com/tailscale/tailscale/issues/188#issuecomment-600728643
175        //
176        // And it seems that even sending to a likely-filtered RFC 5737
177        // documentation-only IPv4 range is enough to set up the mapping.
178        // So do that for now. In the future we might want to classify networks
179        // that do and don't require this separately. But for now help it.
180        let documentation_ip = SocketAddr::from((Ipv4Addr::new(203, 0, 113, 1), 12345));
181
182        socket
183            .send_to(
184                b"net_report; see https://github.com/tailscale/tailscale/issues/188",
185                documentation_ip,
186            )
187            .await?;
188        Ok(())
189    }
190}

219    async fn test_hairpin(hairpinning_works: bool) {
220        // Setup fake net_report and reportstate actors, hairpinning interacts with them.
221        let (net_report_tx, mut net_report_rx) = mpsc::channel(32);
222        let net_report_addr = net_report::Addr {
223            sender: net_report_tx,
224            metrics: Default::default(),
225        };
226        let (reportstate_tx, mut reportstate_rx) = mpsc::channel(32);
227        let reportstate_addr = reportgen::Addr {
228            sender: reportstate_tx,
229        };
230
231        // Create hairpin actor
232        let mut actor = Client::new(net_report_addr, reportstate_addr);
233
234        // Hairpinning works by asking the hairpin actor to send a STUN request to our
235        // discovered public address.  If the router returns it hairpinning works.  We
236        // emulate this by binding a random socket which we pretend is our publicly
237        // discovered address.  The hairpin actor will send it a request and we return it
238        // via the inflight channel.
239        let public_sock = UdpSocket::bind_local_v4(0).unwrap();
240        let ipp_v4 = match public_sock.local_addr().unwrap() {
241            SocketAddr::V4(ipp) => ipp,
242            SocketAddr::V6(_) => unreachable!(),
243        };
244        actor.start_check(ipp_v4);
245
246        // This bit is our dummy net_report actor: it handles the inflight request and sends
247        // back the STUN request once it arrives.
248        let dummy_net_report = tokio::spawn(
249            async move {
250                let net_report::Message::InFlightStun(inflight, resp_tx) =
251                    net_report_rx.recv().await.unwrap()
252                else {
253                    panic!("Wrong message received");
254                };
255                resp_tx.send(()).unwrap();
256
257                let mut buf = BytesMut::zeroed(64 << 10);
258                let (count, addr) = public_sock.recv_from(&mut buf).await.unwrap();
259                info!(
260                    addr=?public_sock.local_addr().unwrap(),
261                    %count,
262                    "Forwarding payload to hairpin actor",
263                );
264                let payload = buf.split_to(count).freeze();
265                let txn = stun::parse_binding_request(&payload).unwrap();
266                assert_eq!(txn, inflight.txn);
267
268                if hairpinning_works {
269                    // We want hairpinning to work, send back the STUN request.
270                    inflight.s.send((Duration::new(0, 1), addr)).unwrap();
271                } else {
272                    // We want hairpinning to fail, just wait but do not drop the STUN response
273                    // channel because that would make the hairpin actor detect an error.
274                    info!("Received hairpin request, not sending response");
275                    tokio::time::sleep(HAIRPIN_CHECK_TIMEOUT * 8).await;
276                }
277            }
278            .instrument(info_span!("dummy-net_report")),
279        );
280
281        // Next we expect our dummy reportstate to receive the result.
282        match reportstate_rx.recv().await {
283            Some(reportgen::Message::HairpinResult(val)) => assert_eq!(val, hairpinning_works),
284            Some(msg) => panic!("Unexpected reportstate message: {msg:?}"),
285            None => panic!("reportstate mpsc has no senders"),
286        }
287
288        // Cleanup: our dummy net_report actor should finish
289        dummy_net_report
290            .await
291            .expect("error in dummy net_report actor");
292    }

294    #[tokio::test]
295    #[traced_test]
296    async fn test_client_drop() {
297        // Setup fake net_report and reportstate actors, hairpinning interacts with them.
298        let (net_report_tx, _net_report_rx) = mpsc::channel(32);
299        let net_report_addr = net_report::Addr {
300            sender: net_report_tx,
301            metrics: Default::default(),
302        };
303        let (reportstate_tx, _reportstate_rx) = mpsc::channel(32);
304        let reportstate_addr = reportgen::Addr {
305            sender: reportstate_tx,
306        };
307
308        // Create hairpin actor
309        let mut client = Client::new(net_report_addr, reportstate_addr);
310
311        // Save the addr, drop the client
312        let addr = client.addr.take();
313        drop(client);
314        tokio::task::yield_now().await;
315
316        // Check the actor is gone
317        let ipp_v4 = SocketAddrV4::new(Ipv4Addr::LOCALHOST, 10);
318        match addr.unwrap().send(Message::StartCheck(ipp_v4)) {
319            Err(_) => (),
320            _ => panic!("actor still running"),
321        }
322    }
