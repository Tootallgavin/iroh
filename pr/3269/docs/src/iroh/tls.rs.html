1//! TLS configuration for iroh.
2//!
3//! Currently there are two mechanisms available
4//! - Raw Public Keys, using the TLS extension described in [RFC 7250]
5//! - libp2p-tls, based on <https://github.com/libp2p/specs/blob/master/tls/tls.md>.
6//!
7//! [RFC 7250]: https://datatracker.ietf.org/doc/html/rfc7250
8
9use std::sync::Arc;
10
11use iroh_base::SecretKey;
12use quinn::crypto::rustls::{QuicClientConfig, QuicServerConfig};
13use tracing::warn;
14
15use self::resolver::AlwaysResolvesCert;
16
17pub(crate) mod certificate;
18pub(crate) mod name;
19mod resolver;
20mod verifier;

22/// Maximum amount of TLS tickets we will cache (by default) for 0-RTT connection
23/// establishment.
24///
25/// 8 tickets per remote endpoint, 32 different endpoints would max out the required storage:
26/// ~200 bytes per session + certificates (which are ~387 bytes)
27/// So 8 * 32 * (200 + 387) = 150.272 bytes, assuming pointers to certificates
28/// are never aliased pointers (they're Arc'ed).
29/// I think 150KB is an acceptable default upper limit for such a cache.
30const MAX_TLS_TICKETS: usize = 8 * 32;
31
32/// TLS Authentication mechanism
33#[derive(Default, Debug, Copy, Clone, PartialEq, Eq)]
34pub(crate) enum Authentication {
35    /// Self signed certificates, based on libp2p-tls
36    X509,
37    /// RFC 7250 TLS extension: Raw Public Keys.
38    #[default]
39    RawPublicKey,
40}
41
42/// Configuration for TLS.
43///
44/// The main point of this struct is to keep state that should be kept the same
45/// over multiple TLS sessions the same.
46/// E.g. the `server_verifier` and `client_verifier` Arc pointers are checked to be
47/// the same between different TLS session calls with 0-RTT data in rustls.
48/// This makes sure that's the case.
49#[derive(Debug)]
50pub(crate) struct TlsConfig {
51    pub(crate) auth: Authentication,
52    pub(crate) secret_key: SecretKey,
53    cert_resolver: Arc<AlwaysResolvesCert>,
54    server_verifier: Arc<verifier::ServerCertificateVerifier>,
55    client_verifier: Arc<verifier::ClientCertificateVerifier>,
56    session_store: Arc<dyn rustls::client::ClientSessionStore>,
57}
58
59impl TlsConfig {
60    pub(crate) fn new(auth: Authentication, secret_key: SecretKey) -> Self {
61        let cert_resolver = Arc::new(
62            AlwaysResolvesCert::new(auth, &secret_key).expect("Client cert key DER is valid; qed"),
63        );
64        Self {
65            auth,
66            secret_key,
67            cert_resolver,
68            server_verifier: Arc::new(verifier::ServerCertificateVerifier::new(auth)),
69            client_verifier: Arc::new(verifier::ClientCertificateVerifier::new(auth)),
70            session_store: Arc::new(rustls::client::ClientSessionMemoryCache::new(
71                MAX_TLS_TICKETS,
72            )),
73        }
74    }
75
76    /// Create a TLS client configuration.
77    ///
78    /// If *keylog* is `true` this will enable logging of the pre-master key to the file in the
79    /// `SSLKEYLOGFILE` environment variable.  This can be used to inspect the traffic for
80    /// debugging purposes.
81    pub(crate) fn make_client_config(
82        &self,
83        alpn_protocols: Vec<Vec<u8>>,
84        keylog: bool,
85    ) -> QuicClientConfig {
86        let mut crypto = rustls::ClientConfig::builder_with_provider(Arc::new(
87            rustls::crypto::ring::default_provider(),
88        ))
89        .with_protocol_versions(verifier::PROTOCOL_VERSIONS)
90        .expect("version supported by ring")
91        .dangerous()
92        .with_custom_certificate_verifier(self.server_verifier.clone())
93        .with_client_cert_resolver(self.cert_resolver.clone());
94        crypto.alpn_protocols = alpn_protocols;
95
96        // TODO: enable/disable 0-RTT/storing tickets
97        crypto.resumption = rustls::client::Resumption::store(self.session_store.clone());
98        crypto.enable_early_data = true;
99
100        if keylog {
101            warn!("enabling SSLKEYLOGFILE for TLS pre-master keys");
102            crypto.key_log = Arc::new(rustls::KeyLogFile::new());
103        }
104
105        crypto
106            .try_into()
107            .expect("expected to have a TLS1.3-compatible crypto provider set (hardcoded)")
108    }
109
110    /// Create a TLS server configuration.
111    ///
112    /// If *keylog* is `true` this will enable logging of the pre-master key to the file in the
113    /// `SSLKEYLOGFILE` environment variable.  This can be used to inspect the traffic for
114    /// debugging purposes.
115    pub(crate) fn make_server_config(
116        &self,
117        alpn_protocols: Vec<Vec<u8>>,
118        keylog: bool,
119    ) -> QuicServerConfig {
120        let mut crypto = rustls::ServerConfig::builder_with_provider(Arc::new(
121            rustls::crypto::ring::default_provider(),
122        ))
123        .with_protocol_versions(verifier::PROTOCOL_VERSIONS)
124        .expect("fixed config")
125        .with_client_cert_verifier(self.client_verifier.clone())
126        .with_cert_resolver(self.cert_resolver.clone());
127        crypto.alpn_protocols = alpn_protocols;
128        if keylog {
129            warn!("enabling SSLKEYLOGFILE for TLS pre-master keys");
130            crypto.key_log = Arc::new(rustls::KeyLogFile::new());
131        }
132
133        // must be u32::MAX or 0 (the default). Any other value panics with QUIC
134        // This is specified in RFC 9001: https://www.rfc-editor.org/rfc/rfc9001#section-4.6.1
135        crypto.max_early_data_size = u32::MAX;
136        crypto
137            .try_into()
138            .expect("expected to have a TLS1.3-compatible crypto provider set (hardcoded)")
139    }
140}
