16/// Errors that can occur when handling a STUN packet.
17#[common_fields({
18    backtrace: Option<Backtrace>,
19    #[snafu(implicit)]
20    span_trace: n0_snafu::SpanTrace,
21})]
22#[allow(missing_docs)]
23#[derive(Debug, Snafu)]
24#[non_exhaustive]
25pub enum StunError {
26    /// The STUN message could not be parsed or is otherwise invalid.
27    #[snafu(display("invalid message"))]
28    InvalidMessage {},
29    /// STUN request is not a binding request when it should be.
30    #[snafu(display("not binding"))]
31    NotBinding {},
32    /// STUN packet is not a response when it should be.
33    #[snafu(display("not success response"))]
34    NotSuccessResponse {},
35    /// STUN response has malformed attributes.
36    #[snafu(display("malformed attributes"))]
37    MalformedAttrs {},
38    /// STUN request didn't end in fingerprint.
39    #[snafu(display("no fingerprint"))]
40    NoFingerprint {},
41    /// STUN request had bogus fingerprint.
42    #[snafu(display("invalid fingerprint"))]
43    InvalidFingerprint {},
44}

46/// Generates a binding request STUN packet.
47pub fn request(tx: TransactionId) -> Vec<u8> {
48    let fp = Fingerprint::default();
49    let msg = StunMessageBuilder::new(methods::BINDING, MessageClass::Request)
50        .with_transaction_id(tx)
51        .with_attribute(fp)
52        .build();
53
54    let encoder = MessageEncoderBuilder::default().build();
55    let mut buffer = vec![0u8; 150];
56    let size = encoder.encode(&mut buffer, &msg).expect("invalid encoding");
57    buffer.truncate(size);
58    buffer
59}
60
61/// Generates a binding response.
62pub fn response(tx: TransactionId, addr: SocketAddr) -> Vec<u8> {
63    let msg = StunMessageBuilder::new(methods::BINDING, MessageClass::SuccessResponse)
64        .with_transaction_id(tx)
65        .with_attribute(XorMappedAddress::from(addr))
66        .build();
67
68    let encoder = MessageEncoderBuilder::default().build();
69    let mut buffer = vec![0u8; 150];
70    let size = encoder.encode(&mut buffer, &msg).expect("invalid encoding");
71    buffer.truncate(size);
72    buffer
73}

75// Copied from stun_rs
76// const MAGIC_COOKIE: Cookie = Cookie(0x2112_A442);
77const COOKIE: [u8; 4] = 0x2112_A442u32.to_be_bytes();
78
79/// Reports whether b is a STUN message.
80pub fn is(b: &[u8]) -> bool {
81    b.len() >= stun_rs::MESSAGE_HEADER_SIZE &&
82        b[0]&0b11000000 == 0 && // top two bits must be zero
83        b[4..8] == COOKIE
84}

86/// Parses a STUN binding request.
87pub fn parse_binding_request(b: &[u8]) -> Result<TransactionId, StunError> {
88    let ctx = DecoderContextBuilder::default()
89        .with_validation() // ensure fingerprint is validated
90        .build();
91    let decoder = MessageDecoderBuilder::default().with_context(ctx).build();
92    let (msg, _) = decoder.decode(b).map_err(|_| InvalidMessageSnafu.build())?;
93
94    let tx = *msg.transaction_id();
95    if msg.method() != methods::BINDING {
96        return Err(NotBindingSnafu.build());
97    }
98
99    // TODO: Tailscale sets the software to tailscale, we should check if we want to do this too.
100
101    if msg
102        .attributes()
103        .last()
104        .map(|attr| !attr.is_fingerprint())
105        .unwrap_or_default()
106    {
107        return Err(NoFingerprintSnafu.build());
108    }
109
110    Ok(tx)
111}

113/// Parses a successful binding response STUN packet.
114/// The IP address is extracted from the XOR-MAPPED-ADDRESS attribute.
115pub fn parse_response(b: &[u8]) -> Result<(TransactionId, SocketAddr), StunError> {
116    let decoder = MessageDecoder::default();
117    let (msg, _) = decoder.decode(b).map_err(|_| InvalidMessageSnafu.build())?;
118
119    let tx = *msg.transaction_id();
120    if msg.class() != MessageClass::SuccessResponse {
121        return Err(NotSuccessResponseSnafu.build());
122    }
123
124    // Read through the attributes.
125    // The the addr+port reported by XOR-MAPPED-ADDRESS
126    // as the canonical value. If the attribute is not
127    // present but the STUN server responds with
128    // MAPPED-ADDRESS we fall back to it.
129
130    let mut addr = None;
131    let mut fallback_addr = None;
132    for attr in msg.attributes() {
133        match attr {
134            StunAttribute::XorMappedAddress(a) => {
135                let mut a = *a.socket_address();
136                a.set_ip(a.ip().to_canonical());
137                addr = Some(a);
138            }
139            StunAttribute::MappedAddress(a) => {
140                let mut a = *a.socket_address();
141                a.set_ip(a.ip().to_canonical());
142                fallback_addr = Some(a);
143            }
144            _ => {}
145        }
146    }
147
148    if let Some(addr) = addr {
149        return Ok((tx, addr));
150    }
151
152    if let Some(addr) = fallback_addr {
153        return Ok((tx, addr));
154    }
155
156    Err(MalformedAttrsSnafu.build())
157}

356    #[test]
357    fn test_response() {
358        let txn = |n| TransactionId::from([n; 12]);
359
360        struct Case {
361            tx: TransactionId,
362            addr: IpAddr,
363            port: u16,
364        }
365        let tests = vec![
366            Case {
367                tx: txn(1),
368                addr: "1.2.3.4".parse().unwrap(),
369                port: 254,
370            },
371            Case {
372                tx: txn(2),
373                addr: "1.2.3.4".parse().unwrap(),
374                port: 257,
375            },
376            Case {
377                tx: txn(3),
378                addr: "1::4".parse().unwrap(),
379                port: 254,
380            },
381            Case {
382                tx: txn(4),
383                addr: "1::4".parse().unwrap(),
384                port: 257,
385            },
386        ];
387
388        for tt in tests {
389            let res = response(tt.tx, SocketAddr::new(tt.addr, tt.port));
390            assert!(is(&res));
391            let (tx2, addr2) = parse_response(&res).unwrap();
392            assert_eq!(tt.tx, tx2);
393            assert_eq!(tt.addr, addr2.ip());
394            assert_eq!(tt.port, addr2.port());
395        }
