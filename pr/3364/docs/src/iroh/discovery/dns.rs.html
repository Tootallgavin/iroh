16/// DNS node discovery
17///
18/// When asked to resolve a [`NodeId`], this service performs a lookup in the Domain Name System (DNS).
19///
20/// It uses the [`Endpoint`]'s DNS resolver to query for `TXT` records under the domain
21/// `_iroh.<z32-node-id>.<origin-domain>`:
22///
23/// * `_iroh`: is the record name
24/// * `<z32-node-id>` is the [`NodeId`] encoded in [`z-base-32`] format
25/// * `<origin-domain>` is the node origin domain as set in [`DnsDiscovery::new`].
26///
27/// Each TXT record returned from the query is expected to contain a string in the format `<name>=<value>`.
28/// If a TXT record contains multiple character strings, they are concatenated first.
29/// The supported attributes are:
30/// * `relay=<url>`: The URL of the home relay server of the node
31///
32/// The DNS resolver defaults to using the nameservers configured on the host system, but can be changed
33/// with [`crate::endpoint::Builder::dns_resolver`].
34///
35/// [z-base-32]: https://philzimmermann.com/docs/human-oriented-base-32-encoding.txt
36#[derive(Debug)]
37pub struct DnsDiscovery {
38    origin_domain: String,
39}
40
41impl DnsDiscovery {
42    /// Creates a new DNS discovery.
43    pub fn new(origin_domain: String) -> Self {
44        Self { origin_domain }
45    }
46
47    /// Creates a new DNS discovery using the `iroh.link` domain.
48    ///
49    /// This uses the [`N0_DNS_NODE_ORIGIN_PROD`] domain.
50    ///
51    /// # Usage during tests
52    ///
53    /// For testing it is possible to use the [`N0_DNS_NODE_ORIGIN_STAGING`] domain
54    /// with [`DnsDiscovery::new`].  This would then use a hosted staging discovery
55    /// service for testing purposes.
56    pub fn n0_dns() -> Self {
57        if force_staging_infra() {
58            Self::new(N0_DNS_NODE_ORIGIN_STAGING.to_string())
59        } else {
60            Self::new(N0_DNS_NODE_ORIGIN_PROD.to_string())
61        }
62    }
63}
64
65impl Discovery for DnsDiscovery {
66    fn resolve(
67        &self,
68        ep: Endpoint,
69        node_id: NodeId,
70    ) -> Option<BoxStream<Result<DiscoveryItem, DiscoveryError>>> {
71        let resolver = ep.dns_resolver().clone();
72        let origin_domain = self.origin_domain.clone();
73        let fut = async move {
74            let node_info = resolver
75                .lookup_node_by_id_staggered(&node_id, &origin_domain, DNS_STAGGERING_MS)
76                .await
77                .map_err(|e| DiscoveryError::from_err("dns", e))?;
78            Ok(DiscoveryItem::new(node_info, "dns", None))
79        };
80        let stream = n0_future::stream::once_future(fut);
81        Some(Box::pin(stream))
82    }
83}
